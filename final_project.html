data = pd.read_csv('crimerussia1.csv', sep=";") 
# Мы собрали данные по всем регионам РФ на 2016 год.(Всего 81 (82 фактически) строка, т.к. Ненецкий автономный округ 
# принимаем за часть Архангельской области, А ЮГРА и Ямало-Ненецкий АО за части Тюменской области, то есть данные по ним агрегированны).
# Population - население региона в тыс. человек (среднегодовое)
# Employment - работающее население в тыс. человек (среднегодовое)
# Income - средний доход по региону в тыс. руб. (среднегодовой в месяц)
# Unemployment - официально зарегистрированные безработные в тыс. чел. (среднегодовое значение)
# Cabin - процент населения, проживающих в аварийных жилых помещениях.(среднегодовой показатель)
# Education - количество студентов высших учебных заведений в тыс. человек. (среднегодовой показатель)
# Illness - количество лиц, проходящих принудительное лечение в психиатрии. (среднегодовой показатель)
# Drugs.1 - количество людей в регионе, проходящих лечение от наркозависимости (среднегодовой)
# Alcohol - рейтинг региона из исследования "Трезвая Россия" 
# Drugs - рейтинг региона по наркозависимости
# Crime - количество зарегистрированных преступлений за год
